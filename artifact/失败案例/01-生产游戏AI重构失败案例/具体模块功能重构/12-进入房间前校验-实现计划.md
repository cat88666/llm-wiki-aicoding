```java
/**
 * 1032 进入房间前预检 handler 分发。
 * <p>因 1032 不要求所有空间都注册 handler（如未来新增空间未补 handler），
 * 不强制 requireNonNull，无匹配时返回 null 交由调用方兜底（即只做密码校验，等价改造前）。</p>
 */
public static IEnterRoomBeforeHandler getEnterRoomBeforeHandler(Long tableId) {
    Objects.requireNonNull(tableId, "tableId cannot be null");
    BaseTableInfo tableInfo = BaseTableInfoCache.getTableInfo(tableId);
    if (tableInfo == null) {
        return null;
    }
    GameSpaceTypeEnums gameSpaceType = GameSpaceTypeEnums.getEnum(tableInfo.getSpaceType());
    if (gameSpaceType == null) {
        gameSpaceType = GameSpaceTypeEnums.CLUB;
    }
    return (IEnterRoomBeforeHandler) findHandler(gameSpaceType, CommonProto.USER_BEFORE_ENTER_ROOM).orElse(null);
}
```

```java
package com.dx.game.frame.service.biz.handler.enterroombefore;

import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.enums.factory.GameSpaceTypeEnums;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.model.cache.UserBaseInfo;
import com.dx.game.frame.model.dto.enterroom.EnterRoomBeforeReqDto;
import com.dx.game.frame.proto.cmd.CommonProto;
import com.dx.game.frame.repository.local.cache.UserBaseInfoCache;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.Objects;

@Slf4j
@Component
public class PrivatedEnterRoomBeforeHandler extends BaseEnterRoomBeforeHandler {

    /**
     * R-02：补密码校验基线（保持与改造前所有空间都校验密码的行为一致），再叠加私人房账号禁入校验。
     * R-05：日志文案「私人房德州游戏」→「私人房」，掼蛋工程不再误导。
     */
    @Override
    public void handle(EnterRoomBeforeReqDto req) {
        super.check(req);   // 密码基线
        Long userId = req.getUserId();
        UserBaseInfo userBaseInfo = UserBaseInfoCache.getUserBaseInfo(userId);
        if (userBaseInfo != null && Objects.equals(1, userBaseInfo.getPrivateMajongStatus())) {
            log.warn("玩家账号被禁止进入私人房,请联系代理 userId:{}", userId);
            throw BusinessException.createException(BusinessExceptionTypeEnums.USER_JOIN_GAME_LOCK_STATUS);
        }
    }

    @Override
    public List<GameSpaceTypeEnums> getSpaceType() {
        return List.of(GameSpaceTypeEnums.PRIVATE);
    }

    @Override
    public Integer getRequestTypeId() {
        return CommonProto.USER_BEFORE_ENTER_ROOM.getId();
    }
}
```

```java
package com.dx.game.frame.service.biz.bootstrap.command.enterRoom;

import com.dx.game.frame.common.utils.GsonUtils;
import com.dx.game.frame.common.utils.NetResponseBuilder;
import com.dx.game.frame.common.utils.SendUtil;
import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.model.context.EnterRoomBeforeContext;
import com.dx.game.frame.model.dto.enterroom.EnterRoomBeforeReqDto;
import com.dx.game.frame.model.dto.enterroom.EnterRoomBeforeRespDto;
import com.dx.game.frame.proto.cmd.CommonProto;
import com.dx.game.frame.service.biz.base.BaseUserCheckAuthService;
import com.dx.game.frame.service.biz.base.BaseUserCheckService;
import com.dx.game.frame.service.biz.handler.enterroombefore.IEnterRoomBeforeHandler;
import com.dx.game.frame.service.factory.GameHandleFactory;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * 进入牌桌前协议 1032。
 * <p>R-02：{@link #check} 改走 {@link GameHandleFactory#getEnterRoomBeforeHandler} 空间分发；
 * 未注册空间兜底为只做密码校验（等价改造前）。</p>
 * <p>R-03：{@code needPwd} / {@code roomNo} 内部上下文字段保留计算，
 * 但当前响应 {@link EnterRoomBeforeRespDto} 不返回它们；1032 成功仅代表「密码校验通过且空间禁入校验通过」。</p>
 * <p>R-04：handleCommand / initContext 中 toJson 大对象 info 日志降级 debug。</p>
 */
@Slf4j
@Component
public class EnterRoomBeforeCommandDefault {

    public void handleCommand(EnterRoomBeforeReqDto req) {
        Long tableId = req.getTableId();
        Long userId = req.getUserId();
        try {
            EnterRoomBeforeContext context = initContext(new EnterRoomBeforeContext(req));
            log.info("进入房间前 tableId:{} userId:{} inRoom:{}",
                    tableId, userId, context.getIsInRoomUser());
            log.debug("进入房间前-请求协议明细 tableId:{} userId:{} context:{}",
                    tableId, userId, GsonUtils.toJson(context));
            check(context);
            SendUtil.send(NetResponseBuilder.buildResponse(CommonProto.USER_BEFORE_ENTER_ROOM, new EnterRoomBeforeRespDto(tableId), req));
        } catch (BusinessException e) {
            log.warn("进入房间前-业务异常 tableId:{} userId:{}", tableId, userId, e);
            SendUtil.send(NetResponseBuilder.buildErrorResponse(CommonProto.USER_BEFORE_ENTER_ROOM, e, req));
        } catch (Exception e) {
            log.error("进入房间前-系统异常 tableId:{} userId:{}", tableId, userId, e);
            SendUtil.send(NetResponseBuilder.buildErrorResponse(CommonProto.USER_BEFORE_ENTER_ROOM,
                    BusinessException.createException(BusinessExceptionTypeEnums.SYSTEM_EXCEPTION), req));
        }
    }

    public EnterRoomBeforeContext initContext(EnterRoomBeforeContext context) {
        EnterRoomBeforeReqDto req = context.getRequest();
        Long tableId = req.getTableId();
        Long userId = req.getUserId();
        boolean inRoomUser = BaseUserCheckService.checkUserIsInRoom(tableId, userId);
        context.setIsInRoomUser(inRoomUser);
        // R-03: needPwd 内部上下文保留计算；当前响应不返回该字段
        context.setNeedPwd(BaseUserCheckAuthService.checkUserisNeedPwd(tableId, userId));
        log.debug("进入房间前-初始化完成 tableId:{} userId:{} context:{}",
                tableId, userId, GsonUtils.toJson(context));
        return context;
    }

    /**
     * R-02：协议检查改走 handler 分发。
     * <ul>
     *   <li>CLUB → {@code ClubEnterRoomBeforeHandler} → {@code checkUserPwd}</li>
     *   <li>HALL/RACE → {@code HallEnterRoomBeforeHandler} → {@code checkUserPwd}</li>
     *   <li>PRIVATE → {@code PrivatedEnterRoomBeforeHandler} → {@code checkUserPwd} + {@code privateMajongStatus} 禁入校验</li>
     *   <li>无匹配（如新增空间未补 handler）→ 兜底 {@code checkUserPwd}，等价改造前</li>
     * </ul>
     */
    public void check(EnterRoomBeforeContext context) {
        EnterRoomBeforeReqDto req = context.getRequest();
        IEnterRoomBeforeHandler handler = GameHandleFactory.getEnterRoomBeforeHandler(req.getTableId());
        if (handler != null) {
            handler.handle(req);
        } else {
            log.warn("进入房间前-无匹配空间 handler,走兜底密码校验 tableId:{} userId:{}",
                    req.getTableId(), req.getUserId());
            BaseUserCheckAuthService.checkUserPwd(req.getTableId(), req.getUserId(), req.getTableEntryPassword());
        }
    }
}
```
