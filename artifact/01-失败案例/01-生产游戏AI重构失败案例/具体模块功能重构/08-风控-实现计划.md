> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现 08 风控重构 R-01~R-04，消除 `checkRiskData` 异常静默成"无风险"的 P0、消除 `realtimeCheckRisk` 多次独立读 cache 导致的快照不一致 P0、集中风险优先级与 Reason 映射、收敛 IP/GPS 日志。保持 `BaseTableRiskService` 所有 public 方法签名 100% 兼容。

**Architecture:** Frame 主导。新增 4 个类：`RiskCheckResult`、`RuleFailure`、`RiskTableSnapshot`（model）、`RiskPolicyRegistry`（service）。`BaseTableRiskService` 内部改造：5 条规则各自 try/catch 进 `safeRule`；`checkRiskSitDown` 用 `RiskCheckResult.success` 决定保守拦截；`realtimeCheckRisk` 一次性 `RiskTableSnapshot.capture` 后全程使用；优先级/Reason 委托 `RiskPolicyRegistry`。掼蛋无变更。

**Tech Stack:** Java 11, Spring Boot, Lombok。无外部新依赖。

package com.dx.game.frame.model.dto.risk;

import com.dx.game.frame.constant.enums.risk.RiskTypeEnum;
import lombok.AllArgsConstructor;
import lombok.Getter;

import java.util.Collections;
import java.util.EnumMap;
import java.util.List;
import java.util.Map;
import java.util.Set;

/**
 * 一次风控规则计算的结果。
 * <ul>
 *   <li>{@code success=true}：5 条规则全部执行成功（命中或未命中），{@code riskMap} 内是真实结果。</li>
 *   <li>{@code success=false}：至少一条规则抛异常，{@code failedRules} 含失败项；调用方必须采取保守策略，不可视为"无风险"。</li>
 * </ul>
 */
@Getter
@AllArgsConstructor
public class RiskCheckResult {
    private final Map<RiskTypeEnum, Set<Long>> riskMap;
    private final boolean success;
    private final List<RuleFailure> failedRules;

    public static RiskCheckResult empty() {
        return new RiskCheckResult(new EnumMap<>(RiskTypeEnum.class), true, Collections.emptyList());
    }

    public boolean hasRisk() {
        return riskMap != null && !riskMap.isEmpty();
    }

    public boolean hasFailure() {
        return failedRules != null && !failedRules.isEmpty();
    }
}

package com.dx.game.frame.model.dto.risk;

import com.dx.game.frame.constant.enums.table.TableUserCacheEnum;
import com.dx.game.frame.constant.enums.user.UserStatusEnum;
import com.dx.game.frame.model.cache.BaseTable;
import com.dx.game.frame.model.cache.BaseTableInfo;
import com.dx.game.frame.model.dto.riskdatacollection.RiskDataCollectionDto;
import com.dx.game.frame.repository.local.cache.BaseTableHandCache;
import com.dx.game.frame.repository.local.cache.BaseTableInfoCache;
import com.dx.game.frame.repository.local.cache.BaseTableUserCache;
import com.dx.game.frame.repository.local.cache.UserRiskCache;
import lombok.AllArgsConstructor;
import lombok.Getter;

import java.util.HashSet;
import java.util.List;
import java.util.Set;

/**
 * 风控实时检测一次执行的快照。
 * 一次 capture 内读全所有依赖 cache，后续算法只读 snapshot，避免多次独立读取导致用户集合 / 手牌状态 / riskData 不一致。
 */
@Getter
@AllArgsConstructor
public class RiskTableSnapshot {
    private final Long tableId;
    private final BaseTableInfo tableInfo;
    private final List<RiskDataCollectionDto> riskData;
    /** 当前 PLAYER_USER 集合（已入座玩家） */
    private final Set<Long> playerUserIds;
    /** PLAYER_USER 中 DEFAULT 状态的用户（开局中已参与本手）；开局中这部分不立即站起 */
    private final Set<Long> activePlayerUserIds;
    /** PLAYER + WATCH + 等待入座 等所有可接收通知的 uid */
    private final Set<Long> allUserIds;
    /** TableHand 是否处于开局中（影响是否立即站起） */
    private final boolean handStarted;
    private final long capturedAt;

    public static RiskTableSnapshot capture(BaseTable table) {
        Long tableId = table.getTableId();
        BaseTableInfo tableInfo = BaseTableInfoCache.getTableInfo(tableId);
        List<RiskDataCollectionDto> riskData = UserRiskCache.getTableRiskData(tableId);
        Set<Long> playerUserIds = new HashSet<>(BaseTableUserCache.fetchTableUserIds(tableId, TableUserCacheEnum.PLAYER_USER));
        Set<Long> activePlayerUserIds = new HashSet<>(BaseTableUserCache.fetchTablePlayerUserIds(tableId, UserStatusEnum.DEFAULT));
        Set<Long> allUserIds = new HashSet<>(BaseTableUserCache.fetchAllUidList(tableId));
        boolean handStarted = BaseTableHandCache.isStartIng(tableId);
        return new RiskTableSnapshot(tableId, tableInfo, riskData,
                playerUserIds, activePlayerUserIds, allUserIds, handStarted, System.currentTimeMillis());
    }
}

package com.dx.game.frame.service.biz.risk;

import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.enums.risk.RiskTypeEnum;
import com.dx.game.frame.constant.enums.user.UserStandTypeEnum;
import com.dx.game.frame.constant.enums.userspecify.KickerReasonEnum;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.Objects;

/**
 * 风险策略集中查询：优先级 / KickerReason / StandType / SeatError。
 * 老 {@code BaseTableRiskService} 内的 RISK_PRIORITY_ORDER 私有常量和 getRiskReason 内部委托到本注册表，
 * 新增风险类型只需要在本类内补一处映射，避免散落到多个文件。
 */
@Component
public class RiskPolicyRegistry {

    private static final List<RiskTypeEnum> PRIORITY_ORDER = List.of(
            RiskTypeEnum.REALTIME_VIDEO_RISK,
            RiskTypeEnum.DEVICE_RISK,
            RiskTypeEnum.GPS_NOT_OPEN,
            RiskTypeEnum.GPS_EQUAL,
            RiskTypeEnum.IP_EQUAL
    );

    /** 按检测顺序返回风险类型；命中第一个即作为用户主风险。 */
    public List<RiskTypeEnum> getPriorityOrder() {
        return PRIORITY_ORDER;
    }

    /** 风险类型 → 站起原因。 */
    public KickerReasonEnum getKickerReason(RiskTypeEnum type) {
        if (type == null) {
            return KickerReasonEnum.RISK_STAND;
        }
        switch (type) {
            case REALTIME_VOICE_RISK:
            case REALTIME_VIDEO_RISK:
                return KickerReasonEnum.RISK_VIDEO_STAND;
            case DEVICE_RISK:
                return KickerReasonEnum.RISK_IOS_STAND;
            case GPS_EQUAL:
            case GPS_NOT_OPEN:
                return KickerReasonEnum.RISK_GPS_STAND;
            case IP_EQUAL:
                return KickerReasonEnum.RISK_IP_STAND;
            default:
                return KickerReasonEnum.RISK_STAND;
        }
    }

    /** 风险类型 → 站起类型（视频风控走专用类型）。 */
    public UserStandTypeEnum getStandType(RiskTypeEnum type) {
        if (type == null) {
            return UserStandTypeEnum.RISK_STAND;
        }
        return Objects.equals(RiskTypeEnum.REALTIME_VIDEO_RISK, type)
                ? UserStandTypeEnum.RISK_VIDEO_STAND
                : UserStandTypeEnum.RISK_STAND;
    }

    /** 入座命中风险时返回给前端的业务错误码（兜底 TABLE_RISK_ERROR）。 */
    public BusinessExceptionTypeEnums getSeatError(RiskTypeEnum type) {
        return type == null ? BusinessExceptionTypeEnums.TABLE_RISK_ERROR : type.getSeatError();
    }
}

import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections.CollectionUtils;

import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.Collections;

import java.util.List;

import java.util.stream.Collectors;

/**
 * 牌桌风控。
 * <p>静态门面：所有 public 方法签名与改造前 100% 兼容。</p>
 * <p>R-01: 5 条规则各自 try/catch，结果以 {@link RiskCheckResult} 返回，
 *    {@link #checkRiskData(Long, List)} 标记 @Deprecated 后委托新方法。</p>
 * <p>R-02: {@link #realtimeCheckRisk(BaseTable)} 一次 capture {@link RiskTableSnapshot} 后全程基于快照。</p>
 * <p>R-03: 优先级 / KickerReason / StandType 委托 {@link RiskPolicyRegistry}。</p>
 * <p>R-04: IP / GPS / 全 riskMap payload 日志降级 debug。</p>
 */
@Slf4j
@Component
public class BaseTableRiskService {

    private static volatile RiskPolicyRegistry policyHolder;

    private static RiskPolicyRegistry policy() {
        RiskPolicyRegistry holder = policyHolder;
        if (holder == null) {
            holder = SpringContextUtils.getBean(RiskPolicyRegistry.class);
            policyHolder = holder;
        }
        return holder;
    }

    /**
     * 入座风控检测。规则失败采取保守拦截（抛 TABLE_RISK_ERROR）。
     */
    public static void checkRiskSitDown(Long tableId, Long userId) {
        List<RiskDataCollectionDto> sitDownRiskData = UserRiskCache.getSitDownRiskData(tableId, userId);
        RiskCheckResult result = checkRiskDataResult(tableId, sitDownRiskData);
        if (!result.isSuccess()) {
            log.warn("入座风控规则部分失败,保守拦截 tableId:{} userId:{} failures:{}",
                    tableId, userId, result.getFailedRules());
            throw BusinessException.createException(BusinessExceptionTypeEnums.TABLE_RISK_ERROR);
        }
        RiskTypeEnum riskTypeEnum = getUserRiskType(userId, result.getRiskMap());
        log.info("入座风控检测结果 tableId:{} userId:{} riskTypeEnum:{}", tableId, userId, riskTypeEnum);
        log.debug("入座风控检测明细 tableId:{} userId:{} sitDownRiskData:{}", tableId, userId, GsonUtils.toJson(sitDownRiskData));
        if (riskTypeEnum != RiskTypeEnum.CHECK_NO_RISK) {
            log.warn("入座风控检测失败 tableId:{} userId:{} riskTypeEnum:{}", tableId, userId, riskTypeEnum);
            throw BusinessException.createMessageException(riskTypeEnum.getSeatError(), riskTypeEnum.getSeatDesc());
        }
    }

    /**
     * R-01 新增：每条规则独立 try/catch，异常不再静默成"无风险"。
     */
    public static RiskCheckResult checkRiskDataResult(Long tableId, List<RiskDataCollectionDto> riskData) {
        Map<RiskTypeEnum, Set<Long>> riskMap = new EnumMap<>(RiskTypeEnum.class);
        List<RuleFailure> failures = new ArrayList<>();

        safeRule(RiskTypeEnum.REALTIME_VIDEO_RISK, () -> UserRiskCache.getRealTimeVideoRiskData(tableId, riskData), riskMap, failures);
        safeRule(RiskTypeEnum.DEVICE_RISK,         () -> UserRiskCache.getDeviceRiskData(tableId, riskData),         riskMap, failures);
        safeRule(RiskTypeEnum.GPS_NOT_OPEN,        () -> UserRiskCache.getGpsSwitchRiskData(tableId, riskData),      riskMap, failures);
        safeRule(RiskTypeEnum.GPS_EQUAL,           () -> UserRiskCache.getGpsRiskData(tableId, riskData),            riskMap, failures);
        safeRule(RiskTypeEnum.IP_EQUAL,            () -> UserRiskCache.getIPRiskData(tableId, riskData),             riskMap, failures);

        boolean success = failures.isEmpty();
        if (!success) {
            log.warn("通用风控检测部分规则失败 tableId:{} failures:{}", tableId, failures);
        }
        log.debug("通用风控检测明细 tableId:{} success:{} riskMap:{} riskData:{}",
                tableId, success, GsonUtils.toJson(riskMap), GsonUtils.toJson(riskData));
        return new RiskCheckResult(riskMap, success, failures);
    }

    private static void safeRule(RiskTypeEnum rule, Supplier<Set<Long>> supplier,
                                 Map<RiskTypeEnum, Set<Long>> sink, List<RuleFailure> failures) {
        try {
            Set<Long> hit = supplier.get();
            if (CollectionUtils.isNotEmpty(hit)) {
                sink.put(rule, hit);
            }
        } catch (Exception e) {
            log.error("风控规则计算异常 rule:{} error:{}", rule, e.getMessage(), e);
            failures.add(new RuleFailure(rule, e.getMessage()));
        }
    }

    /**
     * @deprecated 使用 {@link #checkRiskDataResult(Long, List)} 取得 {@link RiskCheckResult}，
     *             以便区分"规则失败"和"无风险"。本方法保留供过渡，将异常静默成空 map。
     */
    @Deprecated
    public static Map<RiskTypeEnum, Set<Long>> checkRiskData(Long tableId, List<RiskDataCollectionDto> riskData) {
        return checkRiskDataResult(tableId, riskData).getRiskMap();
    }

    /**
     * 获取用户的风控类型（按 {@link RiskPolicyRegistry#getPriorityOrder()} 顺序找到第一个命中）。
     */
    public static RiskTypeEnum getUserRiskType(Long userId, Map<RiskTypeEnum, Set<Long>> riskResults) {
        if (MapUtils.isEmpty(riskResults)) {
            return RiskTypeEnum.CHECK_NO_RISK;
        }
        return policy().getPriorityOrder().stream()
                .filter(type -> riskResults.getOrDefault(type, Collections.emptySet()).contains(userId))
                .findFirst()
                .orElse(RiskTypeEnum.CHECK_NO_RISK);
    }

    /**
     * 牌桌风控实时检测。R-02：整次执行基于 {@link RiskTableSnapshot}，
     * 不在中间再次读 cache，避免用户集合 / 手牌状态 / riskData 不一致。
     */
    public static void realtimeCheckRisk(BaseTable table) {
        Long tableId = table.getTableId();
        try {
            table.setLastRiskCheckTime(System.currentTimeMillis());
            boolean isOldRisked = table.getIsRiskControl();

            RiskTableSnapshot snapshot = RiskTableSnapshot.capture(table);
            RiskCheckResult result = checkRiskDataResult(tableId, snapshot.getRiskData());
            Map<RiskTypeEnum, Set<Long>> riskUserMap = result.getRiskMap();
            table.setIsRiskControl(result.hasRisk());

            if (!isOldRisked && !result.hasRisk() && !result.hasFailure()) {
                return;
            }

            Set<Long> riskUserIds = riskUserMap.values().stream().flatMap(Set::stream).collect(Collectors.toSet());
            List<Long> needStandUsers = computeNeedStandUserIds(snapshot, riskUserIds);

            Map<RiskTypeEnum, Set<Long>> lastRiskUserMap;
            if (!CollectionUtils.isEmpty(needStandUsers)) {
                for (Long standUserId : needStandUsers) {
                    RiskTypeEnum riskTypeEnum = getUserRiskType(standUserId, riskUserMap);
                    UserStandTypeEnum standTypeEnum = policy().getStandType(riskTypeEnum);
                    KickerReasonEnum kickerReasonEnum = policy().getKickerReason(riskTypeEnum);
                    UserLeaveBootstrapService.userStand(standUserId, tableId, standTypeEnum, kickerReasonEnum);
                    log.info("牌桌风控实时检测,风控站起 tableId:{} userId:{} standType:{} reason:{}",
                            tableId, standUserId, standTypeEnum, kickerReasonEnum);
                }
                log.debug("牌桌风控实时检测,站起前明细 tableId:{} needStandUsers:{} riskUserMap:{}",
                        tableId, needStandUsers, GsonUtils.toJson(riskUserMap));
                // 二次确认仍基于 snapshot.riskData，保持原有"看同一份数据"的行为
                lastRiskUserMap = checkRiskDataResult(tableId, snapshot.getRiskData()).getRiskMap();
                table.setIsRiskControl(MapUtils.isNotEmpty(lastRiskUserMap));
                Set<Long> lastRiskUserIds = lastRiskUserMap.values().stream().flatMap(Set::stream).collect(Collectors.toSet());
                Set<Long> standAndNoRiskUsers = needStandUsers.stream()
                        .filter(uid -> !lastRiskUserIds.contains(uid))
                        .collect(Collectors.toSet());
                riskUserMap.forEach((riskType, ids) -> {
                    Set<Long> filtered = ids.stream().filter(standAndNoRiskUsers::contains).collect(Collectors.toSet());
                    if (!filtered.isEmpty()) {
                        lastRiskUserMap.computeIfAbsent(riskType, k -> new HashSet<>()).addAll(filtered);
                    }
                });
            } else {
                lastRiskUserMap = riskUserMap;
            }
            log.debug("牌桌风控实时检测,通知前明细 tableId:{} needStandUsers:{} riskUserMap:{}",
                    tableId, needStandUsers, GsonUtils.toJson(riskUserMap));
            sendCheckRiskNotice(snapshot, table.getIsRiskControl(), needStandUsers, lastRiskUserMap);
        } catch (Exception e) {
            log.error("检测风控发生异常 tableId:{}", tableId, e);
        }
    }

    /**
     * R-02：基于 snapshot 计算需要站起的用户，不重读 cache。
     */
    private static List<Long> computeNeedStandUserIds(RiskTableSnapshot snapshot, Set<Long> riskUserIds) {
        if (CollectionUtils.isEmpty(riskUserIds)) {
            return new ArrayList<>();
        }
        Set<Long> candidates = new HashSet<>(snapshot.getPlayerUserIds());
        if (snapshot.isHandStarted()) {
            candidates.removeAll(snapshot.getActivePlayerUserIds());
        }
        List<Long> neetStandUserIds = candidates.stream()
                .filter(riskUserIds::contains)
                .distinct()
                .collect(Collectors.toList());
        log.debug("获取需要站起的用户 tableId:{} neetStandUserIds:{} candidates:{} riskUserIds:{}",
                snapshot.getTableId(), neetStandUserIds, candidates, riskUserIds);
        return neetStandUserIds;
    }

    /**
     * R-02：基于 snapshot 发送 14038 通知，接收人来自 snapshot.allUserIds + 当次 standUserIdList。
     */
    private static void sendCheckRiskNotice(RiskTableSnapshot snapshot, Boolean isRiskControl,
                                            List<Long> standUserIdList, Map<RiskTypeEnum, Set<Long>> riskUserMap) {
        Long tableId = snapshot.getTableId();
        Set<Integer> tableRiskTypes = MapUtils.isNotEmpty(riskUserMap)
                ? riskUserMap.keySet().stream().map(RiskTypeEnum::getCode).collect(Collectors.toCollection(HashSet::new))
                : Collections.emptySet();

        Set<Long> recipients = new HashSet<>(snapshot.getAllUserIds());
        recipients.addAll(standUserIdList);
        for (Long userId : recipients) {
            RiskTypeEnum typeEnum = getUserRiskType(userId, riskUserMap);
            boolean isStand = standUserIdList.contains(userId);
            String riskTypeDesc = RiskBootstrapService.riskTypeEnumToDesc(tableId, isStand, typeEnum);
            UserRiskNoticeDto noticeDto = new UserRiskNoticeDto();
            noticeDto.setUserId(userId);
            noticeDto.setRiskType(typeEnum.getCode());
            noticeDto.setRiskTypeDesc(riskTypeDesc);
            noticeDto.setTableHasRisk(isRiskControl);
            noticeDto.setTableRiskTypes(new HashSet<>(tableRiskTypes));
            noticeDto.setUserHasRisk(!RiskTypeEnum.CHECK_NO_RISK.equals(typeEnum));
            NetResponseBody netResponseBody = NetResponseBuilder.buildNotice(CommonProto.RISK_CHANGE_NOTICE, noticeDto, tableId);
            SendUtil.sendNotice(userId, netResponseBody);
        }
        log.info("风控检测通知发送结束 tableId:{} isRiskControl:{} recipientCount:{} standCount:{} riskTypeCount:{}",
                tableId, isRiskControl, recipients.size(), standUserIdList.size(), tableRiskTypes.size());
        log.debug("风控检测通知明细 tableId:{} allUidList:{} standUserIdList:{} riskUserMap:{}",
                tableId, GsonUtils.toJson(recipients), GsonUtils.toJson(standUserIdList), GsonUtils.toJson(riskUserMap));
    }

    /**
     * R-03：风控原因映射委托 {@link RiskPolicyRegistry}；方法签名保持。
     */
    public static KickerReasonEnum getRiskReason(RiskTypeEnum riskTypeEnum) {
        return policy().getKickerReason(riskTypeEnum);
    }
}
