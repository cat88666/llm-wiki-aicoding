# 风控数据上报重构 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现风控数据上报重构 R-01~R-05，消除跨桌缓存串用 P0 bug、解耦 1040 主响应与 Redis 同步、修复 1041 GPS 空值 NPE、统一 1041 与 08-风控 规则覆盖，并把 1040/1041 含 IP/GPS 的 info 日志降为 debug。

**Architecture:** Frame 主导，掼蛋无新增类。`UserRiskCache` 内部 cache key 由 `Long` 改为 嵌套静态类 `RiskDataCacheKey(tableId, userId)`；新增 `RiskDataRedisSyncService`（Redis 同步隔离）、`RiskGraphAssembler`（1041 装配 + 风险规则统一）；两个 Command Default 委托新组件，仅留薄壳。`UserRiskInfoDto` 新增可选字段 `riskTypeList` 反映全规则用户级风险。

**Tech Stack:** Java 11, Spring Boot, Lombok, Guava Cache。无外部新依赖。

```java
package com.dx.game.frame.service.biz.risk;

import com.dx.game.frame.common.constant.GameConstant;
import com.dx.game.frame.common.utils.GameUtils;
import com.dx.game.frame.model.cache.BaseTableInfo;

import com.dx.game.frame.repository.redis.service.table.user.HallTableUserSeatRedisService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;

/**
 * 1040 风控数据 Redis 同步：与主响应解耦，异常隔离不上抛。
 */
@Slf4j
@Component
public class RiskDataRedisSyncService {

    @Resource
    private HallTableUserSeatRedisService hallTableUserSeatRedisService;

    /**
     * 安全同步：tableInfo/riskData 任一缺失时跳过；Redis 异常 warn 但不上抛。
     */
    public void syncSafely(Long tableId, Long userId, RiskDataCollectionDto riskData) {
        if (riskData == null) {
            log.warn("syncRiskData-riskData为空，跳过同步 tableId:{} userId:{}", tableId, userId);
            return;
        }
        BaseTableInfo tableInfo = BaseTableInfoCache.getTableInfo(tableId);
        if (tableInfo == null) {
            log.warn("syncRiskData-tableInfo为空，跳过同步 tableId:{} userId:{}", tableId, userId);
            return;
        }
        try {
            RiskDataDto riskDataDto = new RiskDataDto();
            riskDataDto.setIp(riskData.getIp());
            riskDataDto.setLatitude(riskData.getLatitude());
            riskDataDto.setLongitude(riskData.getLongitude());
            String roomCode = GameUtils.getRoomCode(tableInfo.getRoomCode(), tableInfo.getTableCode(), GameConstant.topGameTypeEnum);
            hallTableUserSeatRedisService.syncRiskData(roomCode, tableId, userId, riskDataDto);
        } catch (Exception e) {
            log.warn("syncRiskData-Redis同步失败，不影响主响应 tableId:{} userId:{}", tableId, userId, e);
        }
    }
}
```

```java
/**
 * 1041 风控关系图装配器：
 * <ul>
 *   <li>R-03 GPS 空值短路，不再 NPE。</li>
 *   <li>R-04 复用 {@link BaseTableRiskService#checkRiskData}，把设备/视频/GPS未开等用户级风险通过 {@code riskTypeList} 暴露给客户端。</li>
 * </ul>

@Slf4j
@Component
public class RiskGraphAssembler {

    public RiskDataGraphRespDto assemble(Long tableId, Long currentUserId) {
        List<RiskDataCollectionDto> tableRiskData = UserRiskCache.getTableRiskData(tableId);
        if (tableRiskData == null) {
            throw BusinessException.createException(BusinessExceptionTypeEnums.TABLE_RISK_ERROR);
        }
        BaseTableInfo tableInfo = BaseTableInfoCache.getTableInfo(tableId);
        if (tableInfo == null) {
            log.warn("assembleRiskGraph-tableInfo缺失 tableId:{}", tableId);
            throw BusinessException.createException(BusinessExceptionTypeEnums.TABLE_RISK_ERROR);
        }

        // 1) 全规则风险（含 IP/GPS/GPS未开/设备/视频）
        Map<RiskTypeEnum, Set<Long>> riskMap = BaseTableRiskService.checkRiskData(tableId, tableRiskData);

        // 2) pairwise 边（IP 相同 / GPS 接近 / GPS 未开），GPS 空值短路
        Map<Long, List<UserRiskInfoWrapperDto>> graphConnect = buildPairwiseEdges(tableRiskData, tableInfo);

        Map<Long, RiskDataCollectionDto> riskDataMap = tableRiskData.stream()
                .collect(Collectors.toMap(RiskDataCollectionDto::getUserId, p -> p, (a, b) -> a));
        Map<Long, UserFriendRemarkDto> friendRemarkMap = UserFriendCache.getFriendRemarkCache(
                currentUserId,
                tableRiskData.stream().map(RiskDataCollectionDto::getUserId).collect(Collectors.toSet()));
        List<BaseUserItem> playUserList = BaseTableUserCache.fetchTableUsers(tableId, TableUserCacheEnum.PLAYER_USER);

        RiskDataGraphRespDto resp = new RiskDataGraphRespDto();
        resp.setIsRiskControl(!riskMap.isEmpty());
        List<UserRiskInfoDto> riskPlayerList = new ArrayList<>();
        for (BaseUserItem item : playUserList) {
            UserBaseInfo baseInfo = UserBaseInfoCache.getUserBaseInfo(item.getUserId());
            UserRiskInfoDto info = new UserRiskInfoDto();
            info.setUserId(item.getUserId());
            info.setNickName(baseInfo == null ? "" : baseInfo.getNickName());
            info.setChairId(BaseTableUserCache.fetchUserChairId(tableId, item.getUserId()));
            info.setRemarkName(friendRemarkMap.getOrDefault(item.getUserId(), new UserFriendRemarkDto()).getUserNameRemark());
            info.setTopImagesUrl(baseInfo == null ? "" : baseInfo.getTopImagesUrl());
            if (graphConnect.containsKey(item.getUserId())) {
                info.setRiskGraphList(graphConnect.get(item.getUserId()));
            }
            RiskDataCollectionDto riskData = riskDataMap.get(item.getUserId());
            if (riskData != null && riskData.getGpsImposeSwitch() != null) {
                info.setGpsImposeSwitch(riskData.getGpsImposeSwitch());
            }
            info.setRiskTypeList(collectUserRiskTypes(item.getUserId(), riskMap));
            riskPlayerList.add(info);
        }
        resp.setRiskPlayerList(riskPlayerList.stream()
                .sorted(Comparator.comparing(UserRiskInfoDto::getChairId))
                .collect(Collectors.toList()));
        return resp;
    }

    private List<Integer> collectUserRiskTypes(Long userId, Map<RiskTypeEnum, Set<Long>> riskMap) {
        return riskMap.entrySet().stream()
                .filter(e -> e.getValue().contains(userId))
                .map(e -> e.getKey().getCode())
                .collect(Collectors.toList());
    }

    private Map<Long, List<UserRiskInfoWrapperDto>> buildPairwiseEdges(
            List<RiskDataCollectionDto> riskData, BaseTableInfo tableInfo) {
        Map<Long, List<UserRiskInfoWrapperDto>> adj = new HashMap<>(riskData.size());
        Integer ipSwitch = tableInfo.getIpImposeSwitch();
        Integer gpsSwitch = tableInfo.getGpsImposeSwitch();
        boolean ipOn  = ipSwitch  != null && ipSwitch  == 2;
        boolean gpsOn = gpsSwitch != null && gpsSwitch == 2;
        Integer limitDistance = tableInfo.getLimitToLessThan();

        for (RiskDataCollectionDto a : riskData) {
            List<UserRiskInfoWrapperDto> resList = new ArrayList<>();
            for (RiskDataCollectionDto b : riskData) {
                if (b.getUserId() == null || b.getUserId().equals(a.getUserId())) {
                    continue;
                }
                UserRiskInfoWrapperDto edge = new UserRiskInfoWrapperDto();
                edge.setUserId(b.getUserId());
                edge.setChairId(BaseTableUserCache.fetchUserChairId(tableInfo.getTableId(), b.getUserId()));

                boolean gpsValid = isGpsValid(a) && isGpsValid(b);
                if (gpsOn && gpsValid && limitDistance != null) {
                    edge.setDistance(GeoCalculator.calculateDistance(
                            a.getLatitude().doubleValue(), a.getLongitude().doubleValue(),
                            b.getLatitude().doubleValue(), b.getLongitude().doubleValue()));
                }

                List<Integer> typeList = new ArrayList<>();
                if (ipOn && a.getIp() != null && a.getIp().equals(b.getIp())) {
                    typeList.add(RiskTypeEnum.IP_EQUAL.getCode());
                }
                if (gpsOn && gpsValid && limitDistance != null
                        && isNearby(a, b, limitDistance)) {
                    typeList.add(RiskTypeEnum.GPS_EQUAL.getCode());
                }
                if (gpsOn && (notGpsOn(a) || notGpsOn(b))) {
                    typeList.add(RiskTypeEnum.GPS_NOT_OPEN.getCode());
                }
                if (!typeList.isEmpty()) {
                    edge.setTypeList(typeList);
                    resList.add(edge);
                }
            }
            adj.put(a.getUserId(), resList);
        }
        return adj;
    }

    private boolean isGpsValid(RiskDataCollectionDto data) {
        return data != null
                && data.getLatitude() != null && data.getLongitude() != null
                && data.getLatitude().compareTo(BigDecimal.ZERO) != 0
                && data.getLongitude().compareTo(BigDecimal.ZERO) != 0;
    }

    private boolean notGpsOn(RiskDataCollectionDto data) {
        return data == null || data.getGpsImposeSwitch() == null || data.getGpsImposeSwitch() != 2;
    }

    private boolean isNearby(RiskDataCollectionDto a, RiskDataCollectionDto b, Integer limitDistance) {
        BigDecimal distance = GeoCalculator.calculateDistance(
                a.getLatitude().doubleValue(), a.getLongitude().doubleValue(),
                b.getLatitude().doubleValue(), b.getLongitude().doubleValue());
        return distance.compareTo(new BigDecimal(limitDistance)) < 0;
    }
}
```

```java
import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.model.dto.riskdatagraph.RiskDataGraphReqDto;
import com.dx.game.frame.model.dto.riskdatagraph.RiskDataGraphRespDto;
import com.dx.game.frame.proto.cmd.CommonProto;
import com.dx.game.frame.service.biz.risk.RiskGraphAssembler;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;

/**
 * 风控数据关系图协议:1041
 */
@Slf4j
@Component
public class RiskDataGraphCommandDefault {

    @Resource
    private RiskGraphAssembler riskGraphAssembler;

    public void handleCommand(RiskDataGraphReqDto req) {
        log.info("风控数据关系图协议:1041 tableId:{} userId:{}", req.getTableId(), req.getUserId());
        try {
            RiskDataGraphRespDto resp = riskGraphAssembler.assemble(req.getTableId(), req.getUserId());
            log.debug("风控数据关系图协议:1041 resp tableId:{} userId:{} playerCount:{}",
                    req.getTableId(), req.getUserId(),
                    resp.getRiskPlayerList() == null ? 0 : resp.getRiskPlayerList().size());
            SendUtil.send(NetResponseBuilder.buildResponse(CommonProto.RISK_DATA_GRAPH, resp, req));
        } catch (BusinessException e) {
            log.warn("风控数据关系图协议:1041 业务异常 tableId:{} userId:{}",
                    req.getTableId(), req.getUserId(), e);
            SendUtil.send(NetResponseBuilder.buildErrorResponse(CommonProto.RISK_DATA_GRAPH, e, req));
        } catch (Exception e) {
            log.error("风控数据关系图协议:1041 系统异常 tableId:{} userId:{}",
                    req.getTableId(), req.getUserId(), e);
            SendUtil.send(NetResponseBuilder.buildErrorResponse(CommonProto.RISK_DATA_GRAPH,
                    BusinessException.createException(BusinessExceptionTypeEnums.SYSTEM_EXCEPTION), req));
        }
    }
}
```
