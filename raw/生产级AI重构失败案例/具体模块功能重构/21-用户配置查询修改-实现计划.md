```java
package com.dx.game.frame.service.biz.userconfig;

import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.model.cache.BaseTableInfo;
import com.dx.game.frame.repository.local.cache.BaseTableInfoCache;
import com.dx.game.frame.repository.redis.service.TableUserRedisService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import java.math.BigDecimal;
import java.util.Map;
import java.util.Optional;

/**
 * 自动补码目标值服务，统一 1029 / 3019 两条路径的保存、清理、解析口径。
 *
 * 设计要点：
 * - 目标值统一保存在 TableUserRedisService（tableId+userId 维度），不是玩家维度。
 * - 关闭自动补码时写 0（Redis 当前无 delete 语义）。
 * - 1029/3019 都要求“正整数”，避免两条路径口径不一致。
 */
@Slf4j
@Component
public class AutoComplementConfigService {

    @Resource
    private TableUserRedisService tableUserRedisService;

    /**
     * 保存自动补码目标值。
     * requestValue == null 直接返回 null（视为未携带）；
     * 桌不存在、桌范围非法、目标值非法时抛 PARAM_EXCEPTION。
     */
    public BigDecimal saveIfPresent(Long tableId, Long userId, BigDecimal requestValue, boolean requireInteger) {
        if (requestValue == null) {
            return null;
        }
        BaseTableInfo tableInfo = BaseTableInfoCache.getTableInfo(tableId);
        if (tableInfo == null) {
            log.warn("自动补码配置-牌桌不存在 tableId:{} userId:{} requestValue:{}", tableId, userId, requestValue);
            throw BusinessException.createException(BusinessExceptionTypeEnums.PARAM_EXCEPTION);
        }
        BigDecimal min = tableInfo.getRealMinBringChipScore();
        BigDecimal max = tableInfo.getRealMaxBringChipScore();
        if (!isValidRange(min, max)) {
            log.warn("自动补码配置-牌桌范围非法 tableId:{} userId:{} min:{} max:{}", tableId, userId, min, max);
            throw BusinessException.createException(BusinessExceptionTypeEnums.PARAM_EXCEPTION);
        }
        if (!isValidRequest(requestValue, requireInteger)) {
            log.warn("自动补码配置-目标值非法 tableId:{} userId:{} requestValue:{} requireInteger:{}",
                    tableId, userId, requestValue, requireInteger);
            throw BusinessException.createException(BusinessExceptionTypeEnums.PARAM_EXCEPTION);
        }
        BigDecimal saveValue = requestValue.max(min).min(max);
        tableUserRedisService.setAutoBringChipScore(tableId, userId, saveValue);
        log.info("自动补码配置-保存 tableId:{} userId:{} requestValue:{} saveValue:{} min:{} max:{}",
                tableId, userId, requestValue, saveValue, min, max);
        return saveValue;
    }

    /** 关闭自动补码后清空目标值（写 0）。 */
    public void clear(Long tableId, Long userId) {
        if (tableId == null || userId == null) {
            return;
        }
        tableUserRedisService.setAutoBringChipScore(tableId, userId, BigDecimal.ZERO);
        log.info("自动补码配置-清空目标值 tableId:{} userId:{}", tableId, userId);
    }

    /** 兼容 3019 旧客户端：configMap.autoBringChipScore / configMap.autoBringChip */
    public BigDecimal resolveFromConfigMap(Map<String, Object> configMap) {
        if (configMap == null || configMap.isEmpty()) {
            return null;
        }
        Object raw = Optional.ofNullable(configMap.get("autoBringChipScore"))
                .orElse(configMap.get("autoBringChip"));
        if (raw == null) {
            return null;
        }
        try {
            return new BigDecimal(String.valueOf(raw));
        } catch (NumberFormatException e) {
            log.warn("自动补码配置-configMap 格式错误 raw:{}", raw, e);
            throw BusinessException.createException(BusinessExceptionTypeEnums.PARAM_EXCEPTION);
        }
    }

    private boolean isValidRange(BigDecimal min, BigDecimal max) {
        return min != null && max != null
                && min.compareTo(BigDecimal.ZERO) > 0
                && max.compareTo(min) >= 0;
    }

    private boolean isValidRequest(BigDecimal value, boolean requireInteger) {
        if (value == null || value.compareTo(BigDecimal.ZERO) <= 0) {
            return false;
        }
        if (requireInteger && value.stripTrailingZeros().scale() > 0) {
            return false;
        }
        return true;
    }
}
```

```java
package com.dx.game.frame.service.biz.userconfig;

import com.dx.game.frame.model.cache.UserConfig;
import com.dx.game.frame.repository.local.cache.UserConfigCache;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * UserConfigCache 的可注入封装。
 *
 * 设计要点：
 * - refresh 返回 boolean，异常吞掉记录日志，调用方可显式判断；
 * - prewriteAutoComplement 仅用于 RPC 前收窄同节点并发窗口；
 * - UserConfigCache 静态门面保留供其它链路（结算、补码、退出等）使用。
 */
@Slf4j
@Component
public class UserConfigCacheService {

    public UserConfig get(Long userId) {
        if (userId == null) {
            return null;
        }
        try {
            return UserConfigCache.getUserCfgCache(userId);
        } catch (Exception e) {
            log.warn("UserConfigCacheService-get 异常 userId:{}", userId, e);
            return null;
        }
    }

    /** 刷新缓存，返回是否成功。失败仅记录日志，不向上抛。 */
    public boolean refresh(Long userId) {
        if (userId == null) {
            return false;
        }
        try {
            UserConfigCache.refreshUserCfgCache(userId);
            return true;
        } catch (Exception e) {
            log.warn("UserConfigCacheService-refresh 异常 userId:{}", userId, e);
            return false;
        }
    }

    /** RPC 前预写自动补码开关，仅用于收窄同节点并发窗口。 */
    public void prewriteAutoComplement(Long userId, Integer autoComplement) {
        if (userId == null || autoComplement == null) {
            return;
        }
        UserConfig userConfig = get(userId);
        if (userConfig == null) {
            log.warn("UserConfigCacheService-prewrite 跳过 缓存为空 userId:{} autoComplement:{}", userId, autoComplement);
            return;
        }
        Integer before = userConfig.getAutoComplement();
        userConfig.setAutoComplement(autoComplement);
        log.info("UserConfigCacheService-RPC前预写 autoComplement userId:{} before:{} after:{}",
                userId, before, autoComplement);
    }
}
```
