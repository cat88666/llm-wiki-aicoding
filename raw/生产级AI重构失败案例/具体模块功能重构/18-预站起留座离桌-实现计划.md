```java
package com.dx.game.frame.service.biz.bootstrap.impl;

import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.enums.userleave.PrepStandStatusEnum;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.model.cache.BaseUserItem;
import com.dx.game.frame.repository.local.cache.BaseTableUserCache;
import com.dx.game.frame.service.biz.bootstrap.UserStandPrepBootstrapService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

/**
 * 预站起服务默认实现。
 * 仅负责把预站起意图写入 BaseUserItem.prepStandStatus，
 * 真正的离开动作由 PrepStandLeaveUserVisitor 在本手 / 本轮结束阶段扫描生成。
 *
 * 立即站起（STAND=1）不在此协议范围，请走 1004 站起协议。
 */
@Slf4j
@Service
public class UserStandPrepBootstrapServiceImpl implements UserStandPrepBootstrapService {

    @Override
    public PrepStandStatusEnum prepStand(Long tableId, Long userId, PrepStandStatusEnum prepStandStatusEnum) {
        if (prepStandStatusEnum == null) {
            throw BusinessException.createException(BusinessExceptionTypeEnums.PARAM_EXCEPTION);
        }
        BaseUserItem userItem = BaseTableUserCache.getUser(tableId, userId);
        if (userItem == null) {
            log.warn("预站起-用户不在牌桌 tableId:{} userId:{}", tableId, userId);
            throw BusinessException.createException(BusinessExceptionTypeEnums.NO_EXIST_TABLE_USER);
        }
        if (prepStandStatusEnum == PrepStandStatusEnum.STAND) {
            log.warn("预站起-暂不支持立即站起（STAND=1），请走 1004 站起 tableId:{} userId:{}", tableId, userId);
            throw BusinessException.createException(BusinessExceptionTypeEnums.PARAM_EXCEPTION);
        }
        Integer oldStatus = userItem.getPrepStandStatus();
        userItem.setPrepStandStatus(prepStandStatusEnum.getCode());
        log.info("预站起-状态写入 tableId:{} userId:{} old:{} new:{}",
                tableId, userId, oldStatus, prepStandStatusEnum.getCode());
        return prepStandStatusEnum;
    }
}
```

```java
package com.dx.game.frame.cmd.command.userholdseat;

import com.dx.game.frame.common.thread.EThreadModel;
import com.dx.game.frame.common.thread.ThreadModel;
import com.dx.game.frame.common.utils.NetResponseBuilder;
import com.dx.game.frame.common.utils.SendUtil;
import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.proto.cmd.CommonProto;
import com.dx.netbolt.common.command.request.RequestCommand;
import com.dx.netbolt.common.dto.BaseNetRequestDto;
import com.dx.netbolt.common.dto.ProtoType;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * 1034 留座离桌占位命令。
 * 当前版本不实现完整留座离桌业务，仅明确返回 NOT_SUPPORT_OPERATION，
 * 避免客户端发包后服务端无任何响应。
 * 完整实现待产品确认后单独立项。
 */
@Slf4j
@Component
@ThreadModel(EThreadModel.PLAYER)
public class UserHoldSeatCommand implements RequestCommand<BaseNetRequestDto> {

    @Override
    public void handleCommand(BaseNetRequestDto req) {
        log.warn("留座离桌-当前版本不支持 tableId:{} userId:{}", req.getTableId(), req.getUserId());
        SendUtil.send(NetResponseBuilder.buildErrorResponse(
                CommonProto.USER_HOLD_SEAT,
                BusinessException.createException(BusinessExceptionTypeEnums.NOT_SUPPORT_OPERATION),
                req));
    }

    @Override
    public ProtoType getType() {
        return CommonProto.USER_HOLD_SEAT;
    }
}
```
