```java
package com.dx.game.frame.service.notice;

import com.dx.game.frame.common.utils.GsonUtils;
import com.dx.game.frame.common.utils.NetResponseBuilder;
import com.dx.game.frame.common.utils.SendUtil;

import com.dx.game.frame.model.cache.BaseUserItem;
import com.dx.game.frame.model.context.UserStandContext;
import com.dx.game.frame.model.dto.notice.UserStandNoticeDto;

import com.dx.game.frame.model.dto.userstand.UserStandRespDto;
import com.dx.game.frame.proto.cmd.CommonProto;
import com.dx.game.frame.repository.local.cache.BaseTableUserCache;
import com.dx.game.frame.repository.local.cache.TableUserWalletCache;
import com.dx.game.frame.service.biz.base.BaseUserTrackService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.Objects;

@Slf4j
@Component
public class UserStandNotice {

    @Autowired
    BaseUserTrackService userTrackService;

    /**
     * 用户站起通知。R-02：钱包读取 NPE 兜底；R-03：日志收敛。
     */
    public void notice(UserStandContext context, UserBringOutResult userBringOutResult, BaseUserItem userItem) {
        Long userId = userItem.getUserId();
        Long tableId = userItem.getTableId();
        log.info("用户站起通知开始 tableId:{} userId:{} standType:{} kickerReason:{}",
                tableId, userId, context.getStandType(), context.getKickerReason());
        log.debug("用户站起通知开始-context tableId:{} userId:{} context:{}",
                tableId, userId, GsonUtils.toJson(context));

        // 响应用户
        if (!context.getIsSystem()) {
            UserStandRespDto userStandRespDto = new UserStandRespDto();
            userStandRespDto.setUserId(userId);
            userStandRespDto.setStandType(context.getStandType().getCode());
            userStandRespDto.setKickerReason(context.getKickerReason().getId());
            SendUtil.send(NetResponseBuilder.buildResponse(CommonProto.USER_STAND, userStandRespDto, context.getRequest()));
        }

        // 站起广播通知消息
        UserStandNoticeDto userStandNoticeDto = new UserStandNoticeDto();
        userStandNoticeDto.setUserId(userId);
        userStandNoticeDto.setTableId(tableId);
        userStandNoticeDto.setChairId(userItem.getChairId());
        userStandNoticeDto.setServletTimestamp(System.currentTimeMillis());
        userStandNoticeDto.setStatus(userItem.getStatus());
        userStandNoticeDto.setChipScore(safeFetchChipScore(tableId, userId));
        userStandNoticeDto.setKickerReason(context.getKickerReason().getId());
        if (Objects.equals(context.getStandType().getCode(), UserStandTypeEnum.HOMEOWER_FORCED_STAND.getCode())) {
            userStandNoticeDto.setSpecifyType(UserStandTypeEnum.HOMEOWER_FORCED_STAND.getCode());
        } else {
            userStandNoticeDto.setSpecifyType(context.getStandType().getCode());
        }
        if (userBringOutResult.isBringOutSuccess()) {
            userStandNoticeDto.setAutoBringOutAmount(userBringOutResult.getBringOutAmount());
            userStandNoticeDto.setBureauServiceFee(userBringOutResult.getBureauServiceFee());
        }
        SendUtil.sendNotice(BaseTableUserCache.fetchAllUidList(tableId),
                NetResponseBuilder.buildNotice(CommonProto.USER_STAND_NOTICE, userStandNoticeDto, tableId));
        log.info("用户站起通知完成 tableId:{} userId:{} chairId:{} standType:{}",
                tableId, userId, userStandNoticeDto.getChairId(), userStandNoticeDto.getSpecifyType());
        log.debug("用户站起通知完成-DTO tableId:{} userId:{} userStandNoticeDto:{}",
                tableId, userId, GsonUtils.toJson(userStandNoticeDto));
    }

    /**
     * R-02：安全读取桌面筹码。{@link TableUserWalletCache#getChipScore} 在
     * UserTableItem / TableUserWallet 缓存缺失时会抛异常；站起主流程已经完成
     * 不能让通知组装因此失败，统一兜底为 0 并 warn 日志，便于排障。
     */
    private long safeFetchChipScore(Long tableId, Long userId) {
        try {
            Long chipScore = TableUserWalletCache.getChipScore(tableId, userId);
            return chipScore == null ? 0L : chipScore;
        } catch (Exception e) {
            log.warn("用户站起通知-读取桌面筹码失败,使用0兜底 tableId:{} userId:{}", tableId, userId, e);
            return 0L;
        }
    }
}
```
