```java
package com.dx.game.frame.repository.local.cache.friendremark;

import com.dx.game.frame.model.dto.userfriend.UserFriendRemarkDto;
import lombok.Getter;

/**
 * 好友备注缓存条目。
 * present=true  → 真实备注（dto 字段填充）
 * present=false → 负缓存（dto 为 null，60秒短TTL，由 fetchedAt 判定过期）
 */
@Getter
public class FriendRemarkCacheEntry {

    private final UserFriendRemarkDto dto;
    private final boolean present;
    private final long fetchedAt;

    private FriendRemarkCacheEntry(UserFriendRemarkDto dto, boolean present, long fetchedAt) {
        this.dto = dto;
        this.present = present;
        this.fetchedAt = fetchedAt;
    }

    public static FriendRemarkCacheEntry present(UserFriendRemarkDto dto, long now) {
        return new FriendRemarkCacheEntry(dto, true, now);
    }

    public static FriendRemarkCacheEntry absent(long now) {
        return new FriendRemarkCacheEntry(null, false, now);
    }

    /** 取 dto，负缓存返回空对象，便于调用方继续 BeanUtils 取字段。 */
    public UserFriendRemarkDto getDtoOrEmpty() {
        return dto != null ? dto : new UserFriendRemarkDto();
    }
}
```

- [ ] **Step 1: 创建文件**

```java
package com.dx.game.frame.repository.local.cache.friendremark;

import com.dx.facade.texas.resp.DxGamePersonalRemarksResp;
import com.dx.game.frame.model.dto.userfriend.UserFriendRemarkDto;
import com.dx.game.frame.rpc.consumer.service.DxPlayerRpcClient;
import com.google.common.cache.CacheBuilder;
import com.google.common.cache.CacheLoader;
import com.google.common.cache.LoadingCache;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.collections.CollectionUtils;
import org.springframework.beans.BeanUtils;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import java.util.*;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.TimeUnit;
import java.util.stream.Collectors;

/**
 * 好友备注本地缓存服务。
 * - 命中：TTL 30 分钟 expireAfterAccess
 * - 负缓存：present=false 标记，TTL 60 秒（fetchedAt 判定）
 * - RPC 异常：不写负缓存，下次仍重试，error 级日志可观测
 * - 写：MQ 或当前节点设置备注覆盖为 present 命中
 */
@Slf4j
@Component
public class FriendRemarkCacheService {

    private static final long NEGATIVE_TTL_MS = 60_000L;

    @Resource
    private DxPlayerRpcClient dxPlayerRpcClient;

    private final LoadingCache<Long, ConcurrentHashMap<Long, FriendRemarkCacheEntry>> cache =
            CacheBuilder.newBuilder()
                    .expireAfterAccess(30, TimeUnit.MINUTES)
                    .build(new CacheLoader<Long, ConcurrentHashMap<Long, FriendRemarkCacheEntry>>() {
                        @Override
                        public ConcurrentHashMap<Long, FriendRemarkCacheEntry> load(Long userId) {
                            return new ConcurrentHashMap<>();
                        }
                    });

    public Map<Long, UserFriendRemarkDto> getRemarks(Long userId, Set<Long> remarkUserIds) {
        if (CollectionUtils.isEmpty(remarkUserIds)) {
            return new HashMap<>();
        }
        ConcurrentHashMap<Long, FriendRemarkCacheEntry> shard = cache.getUnchecked(userId);
        long now = System.currentTimeMillis();

        Set<Long> toFetch = remarkUserIds.stream()
                .filter(uid -> needFetch(shard.get(uid), now))
                .collect(Collectors.toSet());

        if (!toFetch.isEmpty()) {
            try {
                List<DxGamePersonalRemarksResp> dataList =
                        dxPlayerRpcClient.queryFriendRemarkList(userId, new ArrayList<>(toFetch));
                Set<Long> returned = new HashSet<>();
                if (dataList != null) {
                    for (DxGamePersonalRemarksResp resp : dataList) {
                        UserFriendRemarkDto dto = new UserFriendRemarkDto();
                        BeanUtils.copyProperties(resp, dto);
                        shard.put(resp.getRemarkUserId(), FriendRemarkCacheEntry.present(dto, now));
                        returned.add(resp.getRemarkUserId());
                    }
                }
                // 仅 RPC 成功时，对真实"不存在"的 key 写负缓存
                for (Long uid : toFetch) {
                    if (!returned.contains(uid)) {
                        shard.put(uid, FriendRemarkCacheEntry.absent(now));
                    }
                }
            } catch (Exception e) {
                // F-02: 不写负缓存，下次重试；error 级日志可观测
                log.error("获取好友备注RPC异常 userId:{} size:{}", userId, toFetch.size(), e);
            }
        }

        Map<Long, UserFriendRemarkDto> result = new HashMap<>(remarkUserIds.size());
        for (Long uid : remarkUserIds) {
            FriendRemarkCacheEntry entry = shard.get(uid);
            result.put(uid, entry == null ? new UserFriendRemarkDto() : entry.getDtoOrEmpty());
        }
        return result;
    }

    public void updateRemark(Long userId, UserFriendRemarkDto updatedDto) {
        if (updatedDto == null || updatedDto.getRemarkUserId() == null) {
            return;
        }
        ConcurrentHashMap<Long, FriendRemarkCacheEntry> shard = cache.getUnchecked(userId);
        shard.put(updatedDto.getRemarkUserId(),
                FriendRemarkCacheEntry.present(updatedDto, System.currentTimeMillis()));
        log.debug("好友备注缓存更新 userId:{} remarkUserId:{}", userId, updatedDto.getRemarkUserId());
    }

    private boolean needFetch(FriendRemarkCacheEntry entry, long now) {
        if (entry == null) {
            return true;
        }
        if (entry.isPresent()) {
            return false;
        }
        return now - entry.getFetchedAt() > NEGATIVE_TTL_MS;
    }
}
```

```java
package com.dx.game.frame.repository.local.cache;

import com.dx.game.frame.common.utils.SpringContextUtils;
import com.dx.game.frame.model.dto.userfriend.UserFriendRemarkDto;
import com.dx.game.frame.repository.local.cache.friendremark.FriendRemarkCacheService;

import java.util.Map;
import java.util.Set;

/**
 * 好友备注缓存静态门面，签名保留以兼容 20+ 历史调用点。
 * 实际逻辑委托给 {@link FriendRemarkCacheService}。
 */
public class UserFriendCache {

    private static volatile FriendRemarkCacheService delegate;

    private static FriendRemarkCacheService delegate() {
        FriendRemarkCacheService local = delegate;
        if (local == null) {
            synchronized (UserFriendCache.class) {
                local = delegate;
                if (local == null) {
                    local = SpringContextUtils.getBean(FriendRemarkCacheService.class);
                    delegate = local;
                }
            }
        }
        return local;
    }

    public static void updateFriendRemarkCache(Long userId, UserFriendRemarkDto updatedDto) {
        delegate().updateRemark(userId, updatedDto);
    }

    public static Map<Long, UserFriendRemarkDto> getFriendRemarkCache(Long userId, Set<Long> remarkUserIds) {
        return delegate().getRemarks(userId, remarkUserIds);
    }
}
```

> 说明：双重检查锁定（DCL）+ volatile 确保首次访问后 delegate 缓存且线程安全。

```java
package com.dx.game.frame.service.biz.friendremark;

import com.dx.facade.texas.req.DxGamePersonalRemarksAddReq;
import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.model.dto.userfriend.FriendRemarkReqDto;
import com.dx.game.frame.model.dto.userfriend.FriendRemarkRespDto;
import com.dx.game.frame.model.dto.userfriend.UserFriendRemarkDto;
import com.dx.game.frame.repository.local.cache.friendremark.FriendRemarkCacheService;
import com.dx.game.frame.rpc.consumer.service.DxPlayerRpcClient;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import java.util.Objects;

/**
 * 1033 好友备注命令的业务编排：校验 → 写玩家服务 → 更新本地缓存 → 构造响应。
 * 不负责协议接入与异常翻译，那些保留在 {@link com.dx.game.frame.cmd.command.friend.FriendRemarkCommand}。
 */
@Slf4j
@Component
public class FriendRemarkCommandService {

    @Resource
    private DxPlayerRpcClient dxPlayerRpcClient;

    @Resource
    private FriendRemarkCacheService friendRemarkCacheService;

    public FriendRemarkRespDto saveRemark(FriendRemarkReqDto req) {
        Long userId = req.getUserId();
        Long remarkUserId = req.getRemarkUserId();
        if (remarkUserId == null || Objects.equals(remarkUserId, userId)) {
            throw BusinessException.createException(BusinessExceptionTypeEnums.USER_ID_PARAM_ERROR);
        }

        DxGamePersonalRemarksAddReq addReq = new DxGamePersonalRemarksAddReq();
        addReq.setUserId(userId);
        addReq.setRemarkUserId(remarkUserId);
        addReq.setUserNameRemark(req.getUserNameRemark());
        addReq.setUserRemark(req.getUserRemark());
        dxPlayerRpcClient.saveFriendRemark(addReq);

        UserFriendRemarkDto updatedDto = new UserFriendRemarkDto();
        updatedDto.setUserId(userId);
        updatedDto.setRemarkUserId(remarkUserId);
        updatedDto.setUserNameRemark(req.getUserNameRemark());
        updatedDto.setUserRemark(req.getUserRemark());
        friendRemarkCacheService.updateRemark(userId, updatedDto);

        FriendRemarkRespDto resp = new FriendRemarkRespDto();
        resp.setUserId(userId);
        resp.setRemarkUserId(remarkUserId);
        resp.setUserNameRemark(req.getUserNameRemark());
        resp.setUserRemark(req.getUserRemark());
        return resp;
    }
}
```

```java
package com.dx.game.frame.cmd.command.friend;

import com.dx.game.frame.common.thread.EThreadModel;
import com.dx.game.frame.common.thread.ThreadModel;
import com.dx.game.frame.common.utils.NetResponseBuilder;
import com.dx.game.frame.common.utils.SendUtil;
import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.model.dto.userfriend.FriendRemarkReqDto;
import com.dx.game.frame.model.dto.userfriend.FriendRemarkRespDto;
import com.dx.game.frame.proto.cmd.CommonProto;
import com.dx.game.frame.service.biz.friendremark.FriendRemarkCommandService;
import com.dx.netbolt.common.command.request.RequestCommand;


import com.dx.netbolt.common.dto.ProtoType;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;

/**
 * 好友备注协议:1033
 * 协议接入 + 异常翻译 + 响应发送；业务编排委托 {@link FriendRemarkCommandService}。
 */
@Slf4j
@Component
@ThreadModel(EThreadModel.PLAYER)
public class FriendRemarkCommand implements RequestCommand<FriendRemarkReqDto> {

    @Resource
    private FriendRemarkCommandService friendRemarkCommandService;

    @Override
    public ProtoType getType() {
        return CommonProto.USER_FRIEND_REMARK;
    }

    @Override
    public void handleCommand(FriendRemarkReqDto req) {
        log.info("好友备注协议:1033 userId:{} remarkUserId:{} nameLen:{} remarkLen:{}",
                req.getUserId(), req.getRemarkUserId(),
                StringUtils.length(req.getUserNameRemark()),
                StringUtils.length(req.getUserRemark()));
        try {
            FriendRemarkRespDto resp = friendRemarkCommandService.saveRemark(req);
            SendUtil.send(NetResponseBuilder.buildResponse(CommonProto.USER_FRIEND_REMARK, resp, req));
            log.info("好友备注协议:1033 成功 userId:{} remarkUserId:{}",
                    req.getUserId(), req.getRemarkUserId());
        } catch (BusinessException e) {
            log.warn("好友备注协议:1033 业务异常 userId:{} remarkUserId:{}",
                    req.getUserId(), req.getRemarkUserId(), e);
            SendUtil.send(NetResponseBuilder.buildErrorResponse(CommonProto.USER_FRIEND_REMARK, e, req));
        } catch (Exception e) {
            log.error("好友备注协议:1033 系统异常 userId:{} remarkUserId:{}",
                    req.getUserId(), req.getRemarkUserId(), e);
            SendUtil.send(NetResponseBuilder.buildErrorResponse(
                    CommonProto.USER_FRIEND_REMARK,
                    BusinessException.createException(BusinessExceptionTypeEnums.SYSTEM_EXCEPTION), req));
        }
    }
}
```
