# 牌桌状态查询重构 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现 13 牌桌状态查询重构 R-01~R-04：修复 `check()` 大厅/俱乐部分支重复 P0 逻辑 bug；修复解散桌 NPE P0；加固 `tableInfo` 空值兜底；收敛 1025 链路 info 日志。1025 请求/响应 100% 协议兼容。

**Architecture:** Frame 主导，掼蛋无变更。改 2 个文件：`TableStatusContext`（新增内部字段 `disbanded`）、`TableStatusCommandDefault`（整体重写 `handleCommand` / `initContext` / `check` / `refreshUser` / `after`）。

**Tech Stack:** Java 11, Spring Boot, Lombok。无外部新依赖。

> 注：`TableStatusContext.tableStatus` 默认值已经是 `GameStatusEnum.GAME_END`，所以 NPE 本身**不在默认值缺失**——是 disband 桌的下游处理还要解引用 status 字段做 DTO 装配，并访问 tableInfo 等其它已清理对象。R-02 通过 short-circuit + 显式 disbanded 标记彻底避开下游链路。

```java
package com.dx.game.frame.service.biz.bootstrap.command.enterRoom;

import com.dx.enums.DeviceTypeEnum;
import com.dx.game.frame.common.thread.BusinessThreadManager;
import com.dx.game.frame.common.utils.GsonUtils;
import com.dx.game.frame.common.utils.NetResponseBuilder;
import com.dx.game.frame.common.utils.SendUtil;
import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.enums.table.GameStatusEnum;
import com.dx.game.frame.constant.exception.BusinessException;
import com.dx.game.frame.model.cache.BaseTable;
import com.dx.game.frame.model.cache.BaseTableInfo;
import com.dx.game.frame.model.context.TableStatusContext;
import com.dx.game.frame.model.dto.secure.AsdRespDto;
import com.dx.game.frame.model.dto.tablestatus.TableSimpleCfgDto;
import com.dx.game.frame.model.dto.tablestatus.TableStatusReqDto;
import com.dx.game.frame.model.dto.tablestatus.TableStatusRespDto;
import com.dx.game.frame.proto.cmd.CommonProto;
import com.dx.game.frame.repository.local.cache.BaseTableCache;
import com.dx.game.frame.repository.local.cache.BaseTableInfoCache;
import com.dx.game.frame.repository.local.cache.UserBaseInfoCache;
import com.dx.game.frame.repository.local.cache.UserConfigCache;
import com.dx.game.frame.repository.redis.dto.LineUpDto;
import com.dx.game.frame.repository.redis.service.RetryLineUpRedisService;
import com.dx.game.frame.repository.utils.AsdConvert;
import com.dx.game.frame.service.biz.base.BaseUserCheckAuthService;
import com.dx.game.frame.service.biz.base.BaseUserCheckService;
import com.dx.game.frame.service.biz.bootstrap.GameTableBootstrapService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.BeanUtils;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;

/**
 * 牌桌状态查询协议:1025。
 * <p>R-01：{@link #check} 改为互斥分支（HALL/PRIVATE/其余），修复大厅桌 Hall+Club 重复跑、RACE 桌错走 Club 的 P0 逻辑 bug。</p>
 * <p>R-02：解散桌路径 short-circuit，{@code disbanded=true} 时下游链路跳过 check/refreshUser 的桌权限读取与配置预热，
 *    {@link #after} 只回 {@code status=GAME_END} + 空配置，不再因为对已清理 cache 解引用而 NPE。</p>
 * <p>R-03：{@link #after} 对 {@code tableInfo == null} 兜底为空 {@link TableSimpleCfgDto}，不再 NPE。</p>
 * <p>R-04：含 context/req/respDto JSON 的 info 日志降级 debug；info 只保留 {@code tableId/userId/status} 元数据。</p>
 */
@Slf4j
@Component
public class TableStatusCommandDefault {

    @Resource
    RetryLineUpRedisService retryLineUpRedisService;

    public void handleCommand(TableStatusReqDto req) {
        Long tableId = req.getTableId();
        Long userId = req.getUserId();
        try {
            if (DeviceTypeEnum.PC.getId().equals(req.getDeviceType())) {
                throw BusinessException.createException(BusinessExceptionTypeEnums.MJ_NOT_YET_OPEN);
            }

            LineUpDto lineUpDto = retryLineUpRedisService.getUserLineUp(userId);
            if (lineUpDto.isLineUp()) {
                log.info("牌桌状态查询-用户在排队中,拒绝进入 tableId:{} userId:{}", tableId, userId);
                log.debug("牌桌状态查询-排队明细 tableId:{} userId:{} lineUp:{}",
                        tableId, userId, GsonUtils.toJsonNoMask(lineUpDto));
                SendUtil.send(NetResponseBuilder.buildErrorResponse(CommonProto.TABLE_STATUS,
                        BusinessException.createException(BusinessExceptionTypeEnums.HALL_LINE_UP_ERROR), req));
                return;
            }

            log.info("牌桌状态查询协议:1025-开始 tableId:{} userId:{}", tableId, userId);
            TableStatusContext context = initContext(new TableStatusContext(req));
            check(context);
            refreshUser(context);
            after(context);
            log.info("牌桌状态查询协议:1025-完成 tableId:{} userId:{} status:{} disbanded:{}",
                    tableId, userId, context.getTableStatus(), context.isDisbanded());
        } catch (BusinessException e) {
            log.warn("牌桌状态查询协议:1025 业务异常 tableId:{} userId:{}", tableId, userId, e);
            SendUtil.send(NetResponseBuilder.buildErrorResponse(CommonProto.TABLE_STATUS, e, req));
        } catch (Exception e) {
            log.error("牌桌状态查询协议:1025 系统异常 tableId:{} userId:{}", tableId, userId, e);
            SendUtil.send(NetResponseBuilder.buildErrorResponse(CommonProto.TABLE_STATUS,
                    BusinessException.createException(BusinessExceptionTypeEnums.SYSTEM_EXCEPTION), req));
        }
    }

    protected TableStatusContext initContext(TableStatusContext context) {
        TableStatusReqDto req = context.getRequest();
        Long tableId = req.getTableId();
        // R-02：disband 短路，set GAME_END 兜底状态，避免 after() NPE
        if (BaseTableCache.checkDisbandTable(tableId)) {
            context.setDisbanded(true);
            context.setTableStatus(GameStatusEnum.GAME_END);
            log.info("牌桌状态查询-牌桌已解散 tableId:{}", tableId);
            return context;
        }

        BaseTable table = BaseTableCache.getTable(tableId);
        if (table == null) {
            if (!GameTableBootstrapService.init(req.getTableId(), req.getUserId())) {
                throw BusinessException.createException(BusinessExceptionTypeEnums.TABLE_USER_STATUS_ERROR);
            }
            table = BaseTableCache.getTable(tableId);
        }
        if (table != null) {
            context.setTableStatus(GameStatusEnum.getByCode(table.getTableStatus()));
        }
        log.debug("牌桌状态查询-初始化完成 tableId:{} userId:{} context:{}",
                tableId, req.getUserId(), GsonUtils.toJson(context));
        return context;
    }

    /**
     * R-01：改互斥分支，HALL 桌不再重复跑 Hall+Club；CLUB/RACE 桌走 Club 校验（行为同改造前 CLUB）。
     * R-02：disband 桌短路。
     * R-03：tableInfo 缺失短路（避免 NPE，warn 日志能定位）。
     */
    protected void check(TableStatusContext context) {
        if (context.isDisbanded()) {
            return;
        }
        Long tableId = context.getTableId();
        Long userId = context.getRequest().getUserId();
        Integer deviceType = context.getRequest().getDeviceType();
        boolean isInRoomUser = BaseUserCheckService.checkUserIsInRoom(tableId, userId);

        BaseTableInfo tableInfo = BaseTableInfoCache.getTableInfo(tableId);
        if (tableInfo == null) {
            log.warn("牌桌状态查询-tableInfo 缺失,跳过权限校验 tableId:{} userId:{}", tableId, userId);
            return;
        }
        if (tableInfo.isHallTable()) {
            BaseUserCheckAuthService.checkUserCanJoinRoomHall(tableId, userId, isInRoomUser, deviceType);
        } else if (tableInfo.isPrivateTable()) {
            BaseUserCheckAuthService.checkUserCanJoinRoomPrivate(tableId, userId, isInRoomUser, deviceType);
        } else {
            // CLUB / RACE / 兜底：保持改造前 CLUB 行为
            BaseUserCheckAuthService.checkUserCanJoinRoom(tableId, userId, isInRoomUser, deviceType);
        }
    }

    protected void refreshUser(TableStatusContext context) {
        if (context.isDisbanded()) {
            return;
        }
        Long tableId = context.getTableId();
        Long userId = context.getRequest().getUserId();
        if (BaseUserCheckService.checkUserIsInRoom(tableId, userId)) {
            return;
        }
        // 每次加入房间都刷新会员基本信息
        BusinessThreadManager.submit(() -> UserBaseInfoCache.getUserBaseInfo(userId));
        // 缓存下用户的配置数据
        UserConfigCache.getUserCfgCache(userId);
    }

    protected void after(TableStatusContext context) {
        Long tableId = context.getTableId();
        Long userId = context.getRequest().getUserId();
        String encryptKey = context.getRequest().getEncryptKey();

        TableStatusRespDto respDto = new TableStatusRespDto();
        respDto.setStatus(context.getTableStatus().getCode());

        if (context.isDisbanded()) {
            // R-02：解散桌只回 status=GAME_END + 空配置，不读 tableInfo / 不生成 ASD
            respDto.setIsPassword(false);
            respDto.setTableSimpleCfgDto(new TableSimpleCfgDto());
            log.info("牌桌状态查询-返回解散响应 tableId:{} userId:{}", tableId, userId);
            SendUtil.send(NetResponseBuilder.buildResponse(CommonProto.TABLE_STATUS, respDto, context.getRequest()));
            return;
        }

        BaseTableInfo tableInfo = BaseTableInfoCache.getTableInfo(tableId);
        TableSimpleCfgDto tableSimpleCfgDto = new TableSimpleCfgDto();
        if (tableInfo != null) {
            BeanUtils.copyProperties(tableInfo, tableSimpleCfgDto);
            respDto.setIsPassword(BaseUserCheckAuthService.checkUserisNeedPwd(tableInfo.getTableId(), userId));
        } else {
            log.warn("牌桌状态查询-tableInfo 缺失,使用空配置兜底 tableId:{} userId:{}", tableId, userId);
            respDto.setIsPassword(false);
        }
        respDto.setTableSimpleCfgDto(tableSimpleCfgDto);

        if (encryptKey != null && !encryptKey.isEmpty()) {
            AsdRespDto asdDto = AsdConvert.convert(userId, tableId, encryptKey);
            respDto.setAsdDto(asdDto);
        }

        log.info("牌桌状态查询-返回信息 tableId:{} userId:{} status:{}",
                tableId, userId, respDto.getStatus());
        log.debug("牌桌状态查询-返回明细 tableId:{} respDto:{}", tableId, GsonUtils.toJson(respDto));
        SendUtil.send(NetResponseBuilder.buildResponse(CommonProto.TABLE_STATUS, respDto, context.getRequest()));
    }
}
```
