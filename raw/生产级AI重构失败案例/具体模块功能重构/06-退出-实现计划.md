```java
package com.dx.game.frame.service.notice;

import com.dx.game.frame.common.utils.GsonUtils;
import com.dx.game.frame.common.utils.NetResponseBuilder;
import com.dx.game.frame.common.utils.SendUtil;
import com.dx.game.frame.constant.enums.user.UserQuitTypeEnum;
import com.dx.game.frame.model.cache.BaseUserItem;
import com.dx.game.frame.model.context.UserQuitContext;
import com.dx.game.frame.model.dto.notice.UserQuitNoticeDto;
import com.dx.game.frame.model.dto.userbringout.UserBringOutResult;
import com.dx.game.frame.proto.cmd.CommonProto;
import com.dx.game.frame.repository.local.cache.BaseTableUserCache;
import com.dx.game.frame.repository.local.cache.TableUserWalletCache;
import com.dx.game.frame.repository.redis.service.NetGatewayUserStatusRedisService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import java.math.BigDecimal;
import java.util.Objects;
import java.util.Set;

@Slf4j
@Component
public class UserQuitNotice {

    @Resource
    private NetGatewayUserStatusRedisService netGatewayUserStatusRedisService;

    /**
     * 用户退出通知。
     * <p>R-01：userItem 缺失时降级为 chairId=null / status=null，不再 NPE，保持"不在房间的用户 1005 仍返回成功"的现有幂等语义。</p>
     * <p>R-02：钱包读取异常兜底返回 {@link BigDecimal#ZERO}，14008 不被钱包缓存阻断。</p>
     * <p>R-03：含 userItem / context / bringOutResult / DTO 的 toJson 日志全部降级 debug。</p>
     */
    public void notice(UserQuitContext context, boolean isWatch, UserBringOutResult userBringOutResult) {
        Long tableId = context.getRequest().getTableId();
        Long userId = context.getRequest().getUserId();
        BaseUserItem userItem = BaseTableUserCache.getUser(tableId, userId);
        log.info("用户退出通知开始 tableId:{} userId:{} quitType:{} kickerReason:{} userItemPresent:{}",
                tableId, userId, context.getQuitType(), context.getKickerReason(), userItem != null);
        log.debug("用户退出通知开始-context tableId:{} userId:{} userItem:{} context:{} bringOut:{}",
                tableId, userId, GsonUtils.toJson(userItem), GsonUtils.toJson(context), GsonUtils.toJson(userBringOutResult));

        UserQuitNoticeDto userQuitNoticeDto = new UserQuitNoticeDto();
        userQuitNoticeDto.setUserId(userId);
        userQuitNoticeDto.setTableId(tableId);
        // R-01: userItem 为 null 时 chairId/status 兜底为 null，避免 NPE
        Integer chairId = (isWatch || userItem == null) ? null : userItem.getChairId();
        userQuitNoticeDto.setChairId(chairId);
        userQuitNoticeDto.setStatus(userItem == null ? null : userItem.getStatus());
        userQuitNoticeDto.setChipScore(safeFetchChipScore(tableId, userId));
        if (Objects.equals(UserQuitTypeEnum.HOMEOWER_FORCED_KICK.getCode(), context.getQuitType().getCode())) {
            userQuitNoticeDto.setSpecifyType(UserQuitTypeEnum.HOMEOWER_FORCED_KICK.getCode());
        } else if (Objects.equals(UserQuitTypeEnum.FORCE_QUIT_LOOK_LIMIT.getCode(), context.getQuitType().getCode())) {
            userQuitNoticeDto.setSpecifyType(UserQuitTypeEnum.FORCE_QUIT_LOOK_LIMIT.getCode());
        }
        if (userBringOutResult.isBringOutSuccess()) {
            userQuitNoticeDto.setAutoBringOutAmount(userBringOutResult.getBringOutAmount());
            userQuitNoticeDto.setBureauServiceFee(userBringOutResult.getBureauServiceFee());
        }

        if (!context.getIsSystem()) {
            SendUtil.send(NetResponseBuilder.buildResponse(CommonProto.USER_QUIT, userQuitNoticeDto, context.getRequest()));
        }
        Set<Long> allUidList = BaseTableUserCache.fetchAllUidList(tableId);
        allUidList.add(userId);
        SendUtil.sendNotice(allUidList, NetResponseBuilder.buildNotice(CommonProto.USER_QUIT_NOTICE, userQuitNoticeDto, tableId));

        netGatewayUserStatusRedisService.removeUserStatus(userId, tableId);

        log.info("用户退出通知完成 tableId:{} userId:{} chairId:{} quitType:{}",
                tableId, userId, userQuitNoticeDto.getChairId(), userQuitNoticeDto.getSpecifyType());
        log.debug("用户退出通知完成-DTO tableId:{} userId:{} userQuitNoticeDto:{}",
                tableId, userId, GsonUtils.toJson(userQuitNoticeDto));
    }

    /**
     * R-02：安全读取桌面筹码。{@link TableUserWalletCache#getChipScore} 在
     * UserTableItem / TableUserWallet 缓存缺失时会抛异常；退出主流程已经完成
     * 不能让通知组装因此失败，统一兜底为 {@link BigDecimal#ZERO} 并 warn 日志。
     */
    private BigDecimal safeFetchChipScore(Long tableId, Long userId) {
        try {
            BigDecimal chipScore = TableUserWalletCache.getChipScore(tableId, userId);
            return chipScore == null ? BigDecimal.ZERO : chipScore;
        } catch (Exception e) {
            log.warn("用户退出通知-读取桌面筹码失败,使用0兜底 tableId:{} userId:{}", tableId, userId, e);
            return BigDecimal.ZERO;
        }
    }
}
```
