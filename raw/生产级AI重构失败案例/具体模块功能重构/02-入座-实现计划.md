# 入座重构 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现入座重构 R-01~R-06，消除占座入座 NPE（P0）、WATCH→PLAYER 并发迁移（P0）、HALL/RACE 静默无响应、占座通知吞异常卡死、限流顺序、日志膨胀、Handler/Default 重复，并将编排/校验/状态写入三个关注点结构性分离，复用 01-进入房间已建的校验链与表级锁框架。

**Architecture:** Frame 主导。新增 `SitDownValidatorChain`（校验）、`SitDownStateService`（状态+表级锁+占座转入座）、`SitDownFlowService`（编排 init→validate→checkChipScore→checkSentinel→applyState→after）。Guandan 提供两个扩展实现：`GdSitDownValidatorContributor`、`GdSitDownStateService`。`Club/PrivateUserSitDownDefault.handle()` 委托 `SitDownFlowService`，`Hall/RaceUserSitDownDefault.handle()` 显式抛 NOT_SUPPORT。

**Tech Stack:** Java 11, Spring Boot, Lombok，无外部新依赖。

```java
package com.dx.game.frame.service.biz.sitdown;

import com.dx.game.frame.model.context.UserSitDownContext;
import com.dx.game.frame.service.biz.base.BaseUserSitDownCheckService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import java.util.List;

/**
 * 入座统一校验链。
 * 内置 frame 校验（踢出/座位/权限/风控），游戏侧通过 ISitDownValidatorContributor 扩展。
 */
@Slf4j
@Component
public class SitDownValidatorChain {

    @Resource
    private List<ISitDownValidatorContributor> contributors;

    public void validate(UserSitDownContext context) {
        // 内置 frame 校验（等价于原 checkSitDown：踢出 / 座位 / 权限 / 风控）
        BaseUserSitDownCheckService.checkSitDown(context);

        // 游戏侧扩展校验
        for (ISitDownValidatorContributor contributor : contributors) {
            contributor.validate(context);
        }
    }
}
```

```java
package com.dx.game.frame.service.biz.sitdown;

import com.dx.game.frame.model.context.UserSitDownContext;
import com.dx.game.frame.service.biz.bootstrap.command.usersitdown.BaseUserSitDownDefault;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.locks.ReentrantLock;

/**
 * 入座状态写入服务。
 * 表级 ReentrantLock 串行化 WATCH→PLAYER 迁移与占座转入座，
 * 覆盖 OccupyToSitDownTask 由 EventManager 调度时与主动入座的并发窗口（S-01/S-03）。
 */
@Slf4j
@Component
public class SitDownStateService {

    @Resource(required = false)
    private IOccupyResitHandler occupyResitHandler;

    private final ConcurrentHashMap<Long, ReentrantLock> tableLocks = new ConcurrentHashMap<>();

    /** 主动入座：WATCH→PLAYER 迁移 + 占座/入座。 */
    public void applyState(UserSitDownContext context, BaseUserSitDownDefault dataBoot) {
        Long tableId = context.getTableId();
        ReentrantLock lock = tableLocks.computeIfAbsent(tableId, k -> new ReentrantLock());
        lock.lock();
        try {
            dataBoot.handler(context);
        } finally {
            lock.unlock();
        }
    }

    /** 占座转入座/结算重坐：走同一把表级锁。 */
    public void occupyToSitDown(Long tableId, Long userId) {
        ReentrantLock lock = tableLocks.computeIfAbsent(tableId, k -> new ReentrantLock());
        lock.lock();
        try {
            if (occupyResitHandler != null) {
                occupyResitHandler.onOccupyToSitDown(tableId, userId);
            } else {
                log.warn("无 IOccupyResitHandler 实现，占座转入座跳过 tableId:{} userId:{}", tableId, userId);
            }
        } finally {
            lock.unlock();
        }
    }

    /** 桌解散时调用，释放锁对象防止内存泄漏。 */
    public void releaseTableLock(Long tableId) {
        tableLocks.remove(tableId);
    }
}
```

```java
package com.dx.game.frame.service.biz.sitdown;

import com.dx.game.frame.model.context.UserSitDownContext;
import com.dx.game.frame.service.biz.base.BaseUserSitDownCheckService;
import com.dx.game.frame.service.biz.bootstrap.UserSitDownBootstrapService;
import com.dx.game.frame.service.biz.bootstrap.command.usersitdown.BaseUserSitDownDefault;
import com.dx.game.frame.service.factory.GameBaseBootstrapFactory;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

import javax.annotation.Resource;

/**
 * 入座编排服务：init → validate → checkChipScore → checkSentinel → applyState → after。
 * R-04：checkSentinel 移到 checkChipScore 之后，余额不足不消耗限流令牌。
 */
@Slf4j
@Component
public class SitDownFlowService {

    @Resource
    private SitDownValidatorChain validatorChain;
    @Resource
    private SitDownStateService   stateService;

    public void sitDown(UserSitDownContext context) {
        Long tableId = context.getTableId();
        Long userId  = context.getRequest().getUserId();
        log.info("玩家入座 tableId:{} userId:{} chairId:{}", tableId, userId, context.getRequest().getChairId());

        // 1. 初始化（私人审核房 PRIVATE_SCORE 由 GdUserSitDownDefault.init 覆盖）
        UserSitDownBootstrapService.init(context);

        // 2. 校验链：踢出 / 座位 / 权限 / 风控 + 游戏侧扩展
        validatorChain.validate(context);

        // 3. 筹码校验（私人房跳过余额校验）
        BaseUserSitDownDefault dataBoot = GameBaseBootstrapFactory.getDataBoot(BaseUserSitDownDefault.class);
        dataBoot.checkChipScore(context);

        // 4. R-04：限流移到筹码校验之后
        BaseUserSitDownCheckService.checkSentinel(tableId, userId);

        // 5. 状态迁移：表级锁 WATCH→PLAYER + 占座
        stateService.applyState(context, dataBoot);

        // 6. 后置：事件 / 响应 / 通知（guandan UserSitDownBootstrap.after 覆盖保留）
        dataBoot.after(context);
        log.info("玩家入座成功 tableId:{} userId:{}", tableId, userId);
    }
}
```

> `dataBoot` 经 `GameBaseBootstrapFactory.getDataBoot(BaseUserSitDownDefault.class)` 解析为 guandan `UserSitDownBootstrap`（覆盖 `checkChipScore/after`），保留掼蛋占座补码兜底逻辑。
