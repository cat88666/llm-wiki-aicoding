```java
package com.dx.game.frame.service.biz.leave;

import com.dx.game.frame.constant.enums.BusinessExceptionTypeEnums;
import com.dx.game.frame.constant.enums.user.LeaveIntent;
import com.dx.game.frame.constant.enums.user.UserArea;
import com.dx.game.frame.constant.enums.user.UserStatusEnum;
import com.dx.game.frame.model.dto.leave.LeaveDecision;
import com.dx.game.frame.repository.local.cache.BaseTableCache;
import com.dx.game.frame.repository.local.cache.BaseTableUserCache;
import com.dx.game.frame.service.biz.bootstrap.UserLeaveBootstrapService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * 离开状态检查的结构化 API 入口。
 * <p>本类不破坏 {@link UserLeaveBootstrapService#commonCheckUserLeave} 等老 API；
 * 老调用方继续工作，新调用方可选择走 {@link #evaluate} 拿到 {@link LeaveDecision}
 * 以便消费 {@code currentArea} 等公共事实，避免重复读 cache。</p>
 * <p>未来 LeaveRuleChain 专项可以替换内部实现，外部调用方无感。</p>
 */
@Slf4j
@Component
public class LeaveDecisionService {

    public LeaveDecision evaluate(LeaveIntent intent, Long tableId, Long userId) {
        UserArea area = detectArea(tableId, userId);
        BusinessExceptionTypeEnums errorCode = invokeUnderlyingCheck(intent, tableId, userId);
        boolean allowed = BusinessExceptionTypeEnums.SUCCESS == errorCode;
        LeaveDecision decision = LeaveDecision.builder()
                .intent(intent)
                .allowed(allowed)
                .errorCode(errorCode)
                .currentArea(area)
                .evidence(allowed ? null : "underlyingCheck=" + errorCode.name())
                .build();
        log.debug("离开决策 tableId:{} userId:{} intent:{} area:{} allowed:{} errorCode:{}",
                tableId, userId, intent, area, allowed, errorCode);
        return decision;
    }

    /**
     * 一次性快照计算用户当前所在区域。
     * 顺序与 {@code UserLeaveDefault.commonCheckUserLeave} 完全一致，保证语义对齐。
     */
    private UserArea detectArea(Long tableId, Long userId) {
        if (BaseTableCache.checkDisbandTable(tableId)) {
            return UserArea.DISBAND;
        }
        if (BaseTableUserCache.hasWatchUser(tableId, userId)) {
            return UserArea.WATCH;
        }
        if (BaseTableUserCache.hasQuitUser(tableId, userId)) {
            return UserArea.QUIT;
        }
        if (BaseTableUserCache.hasOccupyUser(tableId, userId)) {
            return UserArea.OCCUPY;
        }
        if (BaseTableUserCache.fetchTablePlayerUserIds(tableId,
                UserStatusEnum.SIT_DOWN_WAIT_ONE_HAND,
                UserStatusEnum.SIT_DOWN_WAIT_TWO_HAND,
                UserStatusEnum.SIT_DOWN_WAIT_ONE_ROUND).contains(userId)) {
            return UserArea.SIT_DOWN_WAITING;
        }
        if (BaseTableUserCache.hasPlayerUser(tableId, userId)) {
            return UserArea.PLAYER;
        }
        return UserArea.UNKNOWN;
    }

    private BusinessExceptionTypeEnums invokeUnderlyingCheck(LeaveIntent intent, Long tableId, Long userId) {
        switch (intent) {
            case STAND:
            case QUIT:
                return UserLeaveBootstrapService.commonCheckUserLeave(tableId, userId);
            case BREAK_LINK:
            case PREP_STAND:
                return UserLeaveBootstrapService.commonCheckUserBreakLink(tableId, userId);
            default:
                return BusinessExceptionTypeEnums.SYSTEM_EXCEPTION;
        }
    }
}
```

public BusinessExceptionTypeEnums commonCheckUserLeave(Long tableId, Long userId) {
    log.debug("玩家离开检查-玩家信息 tableId:{} userId:{}", tableId, userId);
    if (BaseTableCache.checkDisbandTable(tableId)) {
        log.debug("玩家离开检查,牌局已解散,tableId:{},userId:{}", tableId, userId);
        return BusinessExceptionTypeEnums.TABLE_STATUS_IS_END;
    }
    // 观察者玩家直接离开
    if (BaseTableUserCache.hasWatchUser(tableId, userId)) {
        log.debug("游戏内入座玩家离开检查,观察者玩家可以离开,tableId:{},userId:{}  ", tableId, userId);
        return BusinessExceptionTypeEnums.SUCCESS;
    }
    // 退出玩家直接离开
    if (BaseTableUserCache.hasQuitUser(tableId, userId)) {
        log.debug("游戏内入座玩家离开检查,退出玩家可以离开,tableId:{},userId:{}  ", tableId, userId);
        return BusinessExceptionTypeEnums.SUCCESS;
    }
    // 占座玩家直接离开
    if (BaseTableUserCache.hasOccupyUser(tableId, userId)) {
        log.debug("游戏内入座玩家离开检查,占座玩家可以离开,tableId:{},userId:{}  ", tableId, userId);
        return BusinessExceptionTypeEnums.SUCCESS;
    }
    // 待入座玩家直接离开
    Set<Long> waitUserItems = BaseTableUserCache.fetchTablePlayerUserIds(tableId, UserStatusEnum.SIT_DOWN_WAIT_ONE_HAND, UserStatusEnum.SIT_DOWN_WAIT_TWO_HAND, UserStatusEnum.SIT_DOWN_WAIT_ONE_ROUND);
    if (waitUserItems.contains(userId)) {
        log.debug("游戏内入座玩家离开检查,等待玩家可以离开,tableId:{},userId:{}  ", tableId, userId);
        return BusinessExceptionTypeEnums.SUCCESS;
    }
    // 入座玩家离开
    return checkSitDownUserRealCanLeave(tableId, userId);
}

public void userStand(Long userId, Long tableId, UserStandTypeEnum standTypeEnum, KickerReasonEnum kickerReason) {
